#include<stdio.h>

void  Insertion_Sort_As(int arr[] ,int size){
    int j;
    if(size <= 1)
    return; // 1 element left
    Insertion_Sort_As( arr , size-1); // Sorting elements stored in arr[n-1]
    //arr[5]--->arr[4]--->arr[3]--->arr[2]---->arr[1]
     int temp =  arr[size-1]; // arr[2] --- temp = arr[1] , j = arr[0] 
     j = size-2;
    while (j >=0 && arr[j] > temp)
    {
     arr[j+1] = arr[j];
     j = j-1;
    }
    arr[j+1] = temp;
 }
 
 // Decending Order

    while (j >=0 && arr[j] < temp)
    {
     arr[j+1] = arr[j];
     j = j-1;

 }
