#include<stdio.h>

int Binary_Search_As(int*,int,int ,int ,int);
int Binary_Search_Ds(int*,int,int ,int ,int);
int Binary_Search_As(int arr[],int size,int key , int l, int r){
 
 int mid;
  if(l <= r){
  
   mid = (l + r)/2;
   if(arr[mid] == key)
   return mid;
   else if(arr[mid] > key)
    return Binary_Search_As(arr,size,key,l,mid -1);
   else
    return Binary_Search_As(arr,size,key,mid+1,r);

 }
 return -1;
}
int Binary_Search_Ds(int arr[],int size,int key , int l, int r){
if(l <= r){
   int mid = (l + r)/2;
   if(arr[mid] == key)
   return mid;
   else if(arr[mid] > key)
    return Binary_Search_Ds(arr,size,key,mid+1,r);
   else
    return Binary_Search_Ds(arr,size,key,l,mid-1);

 }
 return -1;

}
