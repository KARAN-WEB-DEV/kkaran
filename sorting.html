#include<stdio.h>

// Sorting Sort for Ascending Order
void Selection_Sort_As(int arr[],int size){

    int j,temp,min;
for(int i = 0; i < size-1; i++){
     min = i; // assuming arr[i] is smallest.
   for ( j = i+1; j < size; j++)
   {
    if(arr[min] > arr[j]){
        min = j;
    }

   }
   //  Case 1 
//  Elemtent is the smallest so you don't have to perform swapping
//  Case 2
// otherwise perform swapping
   if(i != min){
     temp = arr[i];
        arr[i] = arr[min];
        arr[min] = temp;
   }

}

}
// Selection Sort for descending Order
void Selection_Sort_Ds(int arr[],int size){

    int j,temp,min;
for(int i = size-1; i > 0; i--){
     min = i;
   for ( j = i-1; j >= 0; j--)
   {
    if(arr[min] > arr[j]){
        min = j;
    }

   }
  
   if(i != min){
     temp = arr[i];
        arr[i] = arr[min];
        arr[min] = temp;
   }

}

}
