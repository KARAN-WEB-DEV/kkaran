#include<stdio.h>
#include<string.h>
#include<time.h>
void Selection_Sort_As(char string[][100], int n){
// selection sort
for(int i = 0; i < n-1; i++){
    char temp[100];
    int  min = i; // assuming arr[i] is smallest.
   for ( int j = i+1; j < n; j++)
   {
    if(strcmp(string[j],string[min])< 0){
        // strcmp(string comparison)..(2 argument)
       // < 0 first non - matching char has lower value in str 1 than str 2
       // 0 equal string
       // > 0 first non-maching char has a great value i str 1 than str 2
        min = j;
    }

   }
   if(i != min){
     strcpy(temp,string[i]);
     strcpy(string[i],string[min]);
     strcpy(string[min],temp);
   }

}


}
