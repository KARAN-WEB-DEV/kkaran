// Divide and Conquer
// first element as PIVOT 
// Pointer P move RHS and stop whenever it gets the  element >= pivot.(stop :- +infinity(garbage value))
// Pointer q move LHS  and stop whenever it get the element <= pivot.(stop :- assure stop at pivot)
// 1 complete pass - smaller.... pivot.....larger(not sort)
 // if START didn't cross end just swap START and END. 
 // if START cross END  swap pivot with END.
 #include<stdio.h>
 
 int partition( int arr[],int start,int end){
int pivot = arr[start];
int i = start + 1;
int j = end;
while (i <= j)
{
while (arr[i] <= pivot)
 i++;    

while (arr[j]  > pivot)
j--;

if (i < j)
{
    int temp = arr[i];
    arr[i] =  arr[j];
    arr[j] = temp;
}
}

 int temp = arr[start];
 arr[start] = arr[j];
 arr[j] = temp;

return j;
 } 
  // Quick sort function for partition of array into 2 sub array
void Quick_Sort(int arr[],int  start,int end){
    int partition_index; //pivot element at appropriate index
if(start < end){
  partition_index = partition(arr,start,end);
Quick_Sort(arr,start,partition_index-1); //left subarray
Quick_Sort(arr,partition_index +1, end); //right-subarray
}
}

// Decending
int partition_Ds(int arr[], int start, int end) {
    int pivot = arr[start];
    int i = start + 1;
    int j = end;
    while (i <= j) {
        while (i <= end && arr[i] >= pivot) {// Two comparison - out of range n jaye
          i++;
       }
        while (j > start && arr[j] <= pivot) {
            j--;
        }
        if (i < j) {
            int temp = arr[i];
            arr[i] = arr[j];
            arr[j] = temp;
        }
    }

    int temp = arr[start];
    arr[start] = arr[j];
    arr[j] = temp;

    return j;
}
  // Quick sort function for partition of array into 2 sub array
void Quick_Sort_Ds(int arr[],int  start,int end){
    int partition_index;
if(start < end){
  partition_index = partition_Ds(arr,start,end);
Quick_Sort_Ds(arr,start,partition_index-1); //left subarray
Quick_Sort_Ds(arr,partition_index +1, end); //right-subarray
}
}
