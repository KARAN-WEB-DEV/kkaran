#include <stdio.h>

int len(char str[])
{
       int length = 0, i = 0;
    for (i = 0; str[i] != '\0'; i++)
    {
        length++;
    }
    return length;
}
void concate(char str1[],char str2[],int length1,int length2){
    int i;
  for ( i = 0; i <= length2; i++)
  {
    str1[length1 + i] = str2[i]; 
  }
  

}

int main() {
    char str1[100], str2[100];
    int length1,length2;

    printf("Enter the first string: ");

    scanf("%s", str2);
    
    length1 = len(str1);
