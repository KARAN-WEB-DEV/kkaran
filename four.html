#include<stdio.h>

void MulMatrix(int first[][10],int second[][10],int r1, int c1, int r2, int c2) {
   int i,j,k;
   int result[r1][c2];
   // Initializing elements of matrix result to 0.
   //bcoz each index contain a garbage value
   for (i = 0; i < r1; ++i) {
      for (j = 0; j < c2; ++j) {
         result[i][j] = 0;
      }
   }

   // Multiplying first and second matrices and storing it in result
   for (i = 0; i < r1; ++i) { 
    //first loop  for --- row1 in first
      for (j = 0; j < c2; ++j) {
        //socond loop for --- column2 in second
         for (k = 0; k < c1; ++k) {
            //third loop for --column 1 in
         // first & row in second
         
            result[i][j] += first[i][k] * second[k][j];
         }
      }
   }

// for priniting

    printf("\nThe result of Multiplying two Matrix is :\n");
     for(i = 0; i < r1; i++)
    {
        for (j = 0; j < c2; j++)
        {
            printf("%d\t", result[i][j]);
        }
        printf("\n");        
    }
    
}
