  #include<stdio.h>
 
    //   Ascending order
 void Selection_Sort_As(int arr[],int size,int i){
      int  min_index = i,j;
     if (i == size-1)
     {
        return;  // 1 element left..alredy sorted
     }
    
 for ( j = i + 1; j < size; j++)
 {
    if(arr[min_index] > arr[j])
    min_index = j;
 }
 if(min_index != i)
 {
    int temp = arr[min_index];
    arr[min_index] = arr[i];
    arr[i] = temp;


 }
  Selection_Sort_As( arr, size, i+1);

 }
   // Decending order
 void Selection_Sort_Ds(int arr[],int size,int i){
   
if (i <= 0)
return ;

 int min_index = i,j;
    for ( j = i-1; j >= 0; j--)
    {
       if(arr[min_index] > arr[j])
       
        min_index = j;
       
    }
    if(min_index != i)
 {
    int temp = arr[min_index];
    arr[min_index] = arr[i];
    arr[i] = temp;


}
      Selection_Sort_Ds(arr,size,i-1); 
}
