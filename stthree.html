CREATE A STACK IMPLEMENTING THE OPERATIONS ON THE CHARACTERS NODES (DYNAMICALLY) (PUSH, POP, PEEK, ISEMPTY, ISFULL)

#include <stdio.h>
#include <stdlib.h>

struct stack
{
    char data;
    struct stack *next;
};
struct stack *nn, *top = NULL;

// Stack is empty
int isEmpty() {
    if( top == NULL);
    return 1;
    
    return 0;
}

// Check if the stack is full
int isFull() {
    struct stack *n = (struct stack*)malloc(sizeof(struct stack));
    if (n == NULL)
        return 1;
   
    return 0;
}

// PUSH
char push(char value) {
    nn = (struct stack*)malloc(sizeof(struct stack));

    if (!isFull()) {
        nn->data = value;
        nn->next = top;
        top = nn;
        return value;
    }
    else {
        return '\0';
    }
}

// POP
char pop() {
    char value;
    if (isEmpty()) {
        return '\0';
    }
    value = top->data;
    struct stack *temp = top;
    top = top->next;
    free(temp);
    return value;
}

// PEEK
char peek() {
    if (top == NULL) {
        return '\0';
    }
    return top->data;
}

int main() {
    char choice, value;
    int Empty, Full;
    do {
        printf("1  Push\n2  Pop\n3  Peek\n4  IsEmpty\n5  IsFull\n6  Exit\n");
        printf("Enter your choice: ");
        scanf(" %c", &choice); // Notice the space before %c

        switch (choice) {
            case '1':
                printf("Enter the value to add: ");
                scanf(" %c", &value); // Notice the space before %c
                push(value);
                if (value != '\0') {
                    printf("The value %c is pushed successfully\n", value);
                }
                else {
                    printf("Stack is Full\n");
                }
                break;
            case '2':
                value = pop();
                if (value != '\0') {
                    printf("The value %c is popped successfully\n", value);
                } else {
                    printf("Stack is Empty\n");
                }
                break;
            case '3':
                value = peek();
                if (value != '\0') {
                    printf("The Top-Most character is %c\n", value);
                } else {
                    printf("Stack is Empty\n");
                }
                break;
            case '4':
                Empty = isEmpty();
                if (Empty == 1)
                    printf("Stack is empty\n");
                else {
                    printf("Stack is not empty\n");
                }
                break;
            case '5':
                Full = isFull();
                if (Full == 1)
                    printf("Stack is full\n");
                else {
                    printf("Stack is not full\n");
                }
                break;
            case '6':
                printf("Exit\n");
                return 0;
        }
    } while (choice != '6');

    return 0;
}
