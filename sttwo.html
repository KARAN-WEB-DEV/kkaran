CREATE A STACK IMPLEMENTING THE OPERATIONS ON AN INTEGER NODES (DYNAMICALLY) (PUSH, POP, PEEK, ISEMPTY, ISFULL)

#include <stdio.h>
#include <stdlib.h>
struct stack
{
int data;
struct stack *next;
};
struct stack *nn,*top = NULL;

//  stack is empty
int isEmpty() {
    if( top == NULL)
    return 1;

    return 0;
}

//  the stack is full  
int isFull() {
    struct stack*n=(struct stack*)malloc(sizeof(struct stack));
    if(n = NULL)
    return 1;

     return 0;

}
// PUSH
int push(int value)
{


	nn=(struct stack*)malloc(sizeof(struct stack));

	if(! isFull())
	{
	nn->data = value;
	nn->next = top;
	top = nn;
	return value;
	
	}	
	else
	{
		return -1; // stack Overflow
	
	}


}


//  POP
int pop() {
    int value;
    if (isEmpty()) {
       
        return -1; 
    }
     value = top->data;
    struct stack* temp = top;
    top = top->next;
    free(temp);
    return value ; 
}

//  PEEK 
int peek() {
    if (top == NULL) {
       
        return -1; 
    }
    return top->data;
}



int main() {
    int choice, value;

    do {
           printf("1  Push\n2  Pop\n3  Peek\n4  IsEmpty\n5  Exit\n");
       
        printf("Enter your choice: ");
        scanf("%d", &choice);

        switch (choice) {
            case 1:
                printf("Enter the value to add: ");
                scanf("%d", &value);
                if (push(value) != -1) {
                    printf("The value  %d is pushed successfully\n", value);
                }
                else{
                    printf("\nStack is Full\n");
                }
                break;
           case 2:
    value = pop();
    if (value != -1) {
        printf("The value %d is popped successfully\n", value);
    } else {
        printf("Stack is Empty\n");
    }
    break;

            case 3:
                value = peek();
                if (value != -1) {
                    printf("The Top-Most value is %d\n", value);
                } else {
                    printf("Stack is Empty\n");
                }
                break;
            case 4:
                if (isEmpty()) {
                    printf("Stack is empty\n");
                } else {
                    printf("Stack is not empty\n");
                }
                break;
            case 5:
                printf("Exit\n");
                return 0;
            
        }
    }while(choice != 5);

   

    return 0;
}
