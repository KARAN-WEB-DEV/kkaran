ENTER A INTEGER ARRAY FROM THE USER, ENTER AN INTEGER VALUE AND PERFORM LINEAR SEARCH ALGORITHM THROUGH RECURSION.

#include<stdio.h>
int Linear_Search(int*,int ,int);

int Linear_Search(int arr[],int size,int key){
if (arr[size-1] == key)
return size-1;

if(size == 0)
return -1;
 // Searching On Continution...
 return Linear_Search(arr,size-1,key);
}
 int main(){
    int arr[10];
    int size, key,i,index;

    printf("Enter the size of Array\n");
    scanf("%d",&size);

    printf("Enter the elements in Array\n");
    for( i = 0; i< size ; i++)
    scanf("%d",&arr[i]);

    printf("Enter the key to search\n");
    scanf("%d",&key);

    index = Linear_Search(arr, size,key);
    (index == -1)?printf("\nNot Present")
    :printf("\nPresent at %d index",index);

    return 0;
 }
