CREATE THE PROGRAM TO MATCH THE SINGLE BRACKET USING STACKS.

#include <stdio.h>

int match_char(char a, char b) {
    if ((a == '[' && b == ']') || (a == '{' && b == '}') || (a == '(' && b == ')')) {
        return 1; //  true 
    }
    return 0; //  false 
}

int isFull(char stack[], int size, int top) {
    if (top == size - 1) {
        return 1; 
    }
    return 0; 
}

int isEmpty(char stack[], int top) {
    if (top == -1) {
        return 1;
    }
    return 0;
}

int push(char stack[], int size, char bracket, int top) {
    if (isFull(stack, size, top)) {
        return -1;
    }
    top = top + 1;
    stack[top] = bracket;
    return top;
}

int pop(char stack[], int top, char close_bracket) {
    char open_bracket;
    if (isEmpty(stack, top)) {
        return -100; //  indicae emptiness
    } else {
        open_bracket = stack[top];
        if (match_char(open_bracket, close_bracket)) {
            top--;
            return top;
        } else {
            return -100; // Indicate error
        }
    }
}

int main() {
    int i, top = -1;
    char exp[20], stack[20];
    printf("\nEnter the expression: ");
    gets(exp);

    for (i = 0; exp[i] != '\0'; i++) {
        if (exp[i] == '(' || exp[i] == '[' || exp[i] == '{') {
            top = push(stack, 20, exp[i], top);
        }
        if (exp[i] == ')' || exp[i] == ']' || exp[i] == '}') {
            top = pop(stack, top, exp[i]);
            if (top == -100) {
                break;
            }
        }
    }

    if (top == -1) {
        printf("\nBrackets are matching\n");
    } else {
        printf("\nBrackets aren't matching\n");
    }

    return 0;
}
