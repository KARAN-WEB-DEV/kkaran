//  recursive Merge Sort is a TOP DOWN APPROACH first breakdown of array
//  then sort 
#include<stdio.h>

void Merge_As(int arr[],int l, int r,int mid){
int i,j,k;
 i = l;              // left - half
 j = mid + 1;        // Right half
 k = l;              // temp array index
 int temp[20];
while(i <= mid && j <= r){
 if(arr[i] <= arr[j]){
 temp[k] = arr[i];
 i++; k++;
 }
 else{
    temp[k] = arr[j];
 j++; k++;
 }}
 while (i <= mid)
 {
    temp [k] = arr[i];
    i++;k++;
 }
 while (j <= r)
 {
    temp[k] = arr[j];
    j++;k++;
 }
 // Copying the temp[Elements] in O.G array arr[Elements]..
for ( i = l; i <= r ; i++)
{
    arr[i] = temp[i];
}
}
void Merge_Sort_As(int arr[],int l,int r){
    int mid;
 if(l < r){
    mid = (l + r)/2;
 Merge_Sort_As(arr,l,mid);
 Merge_Sort_As(arr,mid+1,r);
 Merge_As(arr,l,r,mid);

 }

}

 i = l;

}

    mid = (l + r)/2;

}
