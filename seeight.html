CALCULATE THE BEST AND WORST COMPLEXITY OF THE LINEAR SEARCHING ALGORITHM WITH EXACT TIME TAKEN BY THE ALGORITHM ON INTEGER ARRAY.

#include <stdio.h>
#include <time.h>
#include <stdlib.h>

// Linear search function
int linearSearch(int arr[], int n, int key) {
    for (int i = 0; i < n; i++) {
        if (arr[i] == key) {
            return i;  
        }
    }
    return -1; 
}

int main() {
    int n;
    clock_t start, end;
    double timeComplexity;

printf("Enter the size of the array: ");
scanf("%d", &n);    // let say 1000

    int arr[n];
printf("Enter Elements in Array\n");
for ( int i = 0; i < n; i++)
{
    arr[i] = rand();
}

    //  Best-case scenario
    start = clock();
    int index1 = linearSearch(arr, n, arr[0]);
    end = clock();

    timeComplexity = ((double)(end - start)) / CLOCKS_PER_SEC;

 (index1 == -1)?printf("\nNot Present\n")
:printf("\nPresent at %d Index\n",index1);
       printf("Best-case time complexity: O(1) \n");
      printf("Time taken  by Algorithm: %lf seconds\n", timeComplexity);

printf("\n-------------------------------------------------\n");

    // Worst-case scenario
    start = clock();
    int index2 = linearSearch(arr, n, arr[n-1]);
    end = clock();

      timeComplexity = ((double)(end - start)) / CLOCKS_PER_SEC;
      
 (index2 == -1)?printf("\nNot Present\n")
:printf("\nPresent at %d Index\n",index2);
       printf("Worstt-case time complexity: O(n) \n");
         printf("Time taken by Algorithm: %lf seconds\n", timeComplexity);

    return 0;
}
