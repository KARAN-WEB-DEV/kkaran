CREATE A STACK IMPLEMENTING THE OPERATIONS ON AN INTEGER ARRAY (PUSH, POP, PEEK, ISEMPTY, ISFULL)

#include<stdio.h>

int push(int[],int,int,int);
int pop(int[],int);
int peek(int[],int);
int isFull(int[],int,int);
int isEmpty(int[],int);

// -----------is Full or not-----------
int isFull(int stack[],int size,int top){
  if(top == size-1){
  return 1;// true
  }
  return 0;//false
}

// -------------is Empty or not--------
int isEmpty(int stack[],int top){
  if(top == -1)
  {
    return 1;
  }
  return 0;
}
  //------------- Push operation--------
int push(int stack[],int size, int value,int top){
 
  if( isFull(stack,size,top)){
    return -1; 
   
  }
  else{
    top++;
stack[top] = value;
return top;
  }

}
// --------------POP Operation------------
int pop( int stack[],int top){
  
if(isEmpty(stack,top)){
 return -1;
}
else 
{
top--;
return top;
}

}

// --------------PEEK Operation-----------
int peek(int stack[], int top){
 if(isEmpty(stack,top)){
 return -1;
 }
 else
 return top;
}
