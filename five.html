#include<stdio.h>

void CopyMatrix(int first[10][10],int row,int column){
        int Copied[row][column];
     for ( int i = 0; i < row; i++)
     {
        for ( int j = 0; j < column; j++)
        {
            Copied[i][j] = first[i][j];
        }
        printf("\n");
     }
     // Printing Copied Matrix
     printf("Copied MatriX :\n");
     for(int i = 0; i < row; i++)
    {
        for (int j = 0; j < column; j++)
        {

        }
        printf("\n");        
    }
     
    }
