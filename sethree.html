ENTER A STRING FROM THE USER, ENTER A CHARACTER AND PERFORM LINEAR SEARCH ALGORITHM THROUGH ITERATION.

#include <stdio.h>
int linearSearch(char str[], char key) {
    for (int i = 0; str[i] != '\0'; i++) {
        if (str[i] == key) {
            return i;
        }
    }
    return -1; 
}

int main() {
    char str[100];
    char key;

    
    printf("\nEnter a string: \n");
    gets(str); 

    printf("\nEnter a character to search for: \n");
    scanf(" %c", &key);

    int index = linearSearch(str, key);

   (index == -1)?printf("\nNot Present\n")
:printf("\nPresent at %d index\n",index);

    return 0;
}
/*

Output:-
Enter a string:
Hello World

Enter a character to search for:
r

Present at 8 index*/
