#include <stdio.h>

int push(int stack[], int size, int value, int stackNum, int top1, int top2) {
   if(stackNum == 1){
    if (top1 + 1 == top2 || top1 == size-1 || top2 == 0) {
       
        return -1 ; // overflow
    }

    else {
        top1++;
        stack[top1] = value;
        return top1;
    } 
   }

   else{
    if(top2 == top1 + 1 || top1 == size-1 || top2 == 0)
        return -1;
    else   {
        top2--;
        stack[top2] = value;
        return top2;
    }
   }
}

int pop(int stack[],int size, int stackNum, int top1, int top2) {
   if (stackNum == 1){
    if (top1 == -1) {    
        return -1; // underflow
    } 
    else{
        top1 = top1 - 1;
        return top1;
    }
   }

   else{
    if( top2 == size) {
      
        return -1;
    }
else   {
        top2 = top2 +1;
        return top2;
    }
   
}
}
