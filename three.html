Enter two Matrices from user, subtract the matrices and display the answer.

#include<stdio.h>

void subMatrix(int arr1[10][10], int arr2[10][10],int row ,int column){
    int i, j;
    int result[10][10];
    for (i = 0; i < row; i++)
    {
        for (j = 0; j < column; j++)
        {
            result[i][j] = arr1[i][j] - arr2[i][j];
        }
    }

// for priniting

    printf("\nThe result of Subtracting two Matrix is :\n");
     for(i = 0; i < row; i++)
    {
        for (j = 0; j < column; j++)
        {
            printf("%d\t", result[i][j]);
        }
        printf("\n");        
    }
    
}

int main(){
    int arr1[10][10], arr2[10][10], i, j,row,column;
    printf("Enter the number of rows (between 1 and 100): ");
  scanf("%d", &row);
  printf("Enter the number of columns (between 1 and 100): ");
  scanf("%d", &column);
  printf("For First Matrix\n");
    for( i = 0; i < row; i++)
    {
        for (j = 0; j < column; j++)
        {
            
            printf("A[%d][%d]\t", i, j);
            scanf("%d", &arr1[i][j]);
            printf("\n");
        }
    }
    // for 2nd matrix
printf("For Second Matrix\n");
    for( i = 0; i < row; i++)
    {
        for (j = 0; j < column; j++)
        {
            
            printf("A[%d][%d]\t", i, j);
            scanf("%d", &arr2[i][j]);
            printf("\n");
        }
    }

//printing
    printf("Matrix 1: \n");
     for(i = 0; i < row; i++)
    {
        for (j = 0; j < column; j++)
        {
            printf("%d\t", arr1[i][j]);
        }

    printf("Matrix 2: \n");
     for(i = 0; i < row; i++)
    {
        for (j = 0; j < column; j++)
        {
            printf("%d\t", arr2[i][j]);
        }
        printf("\n");        
    }    
    subMatrix(arr1, arr2, row,column);

}
