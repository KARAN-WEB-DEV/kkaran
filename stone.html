CREATE A STACK IMPLEMENTING THE OPERATIONS ON THE CHARACTERS ARRAY (PUSH, POP, PEEK, ISEMPTY, ISFULL)

#include <stdio.h>

int push(char[], int, char, int);
int pop(char[], int);
int peek(char[], int);
int isFull(char[], int, int);
int isEmpty(char[], int);

//  is full
int isFull(char stack[], int size, int top) {
    if (top == size - 1) {
        return 1; // true
    }
    return 0; // false
}

//  is empty
int isEmpty(char stack[], int top) {
    if (top == -1) {
        return 1;
    }
    return 0;
}

// Push operation
int push(char stack[], int size, char value, int top) {
    if( isFull(stack,size,top)){
    return -1; //indicates  unsuccessful push oper...
   
  }
  else{
    top++;
stack[top] = value;
return top;
  }
}

// Pop operation
int pop(char stack[], int top) {
 if(isEmpty(stack,top)){
 return -1;
}
else 
{
top--;
return top;
}
}

// Peek operation
int peek(char stack[], int top) {
    if(isEmpty(stack,top)){
 return -1;
 }
 else
 return top;
}

    char stack[10];
    int size, key, top = -1;

          press = getchar();

    case '1':

            (top == -1)? printf("\nStack is full\n"):
  printf(" The value  %c is pushed successfully\n",stack[top]);
                break;

    case '2':
                top = pop(stack, top);
                (top == -1)? printf("\nStack is Empty\n"):
printf("\nThe value is poped successfully and now TOP pointing to %d index",top);
                break;

case '3':
            top = peek(stack, top);
            (top == -1)?
            printf("\n Stack Is Empty\n"):
            printf("The Top-Most character is %c\n",stack[top]);
